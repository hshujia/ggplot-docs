Usage

Arguments

Name Description
position Position adjustments to points. identity

import numpy as np
import pandas as pd

df = pd.DataFrame({
    "x": range(100),
    "y": np.random.choice([-1, 1], 100)
})
df.y = df.y.cumsum()

ggplot(df, aes(x='x', y='y')) +\
    geom_point()

ggplot(df, aes(x='x', y='y')) +\
    geom_step()
