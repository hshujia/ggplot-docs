title: scale_color_manual {% extends "docpage.html" %} {% block body %}

Usage

scale_color_manual({ x: 'blue', y: 'red', ... })

Arguments

Name Description
values Mapped values for colors.

Examples

 ggplot(aes(x='wt', y='mpg', colour='factor(cyl)'), data=mtcars) +\
    geom_point() +\
    scale_colour_manual(values=["blue", "purple", "orange"])
ggplot(aes(x='price', color='cut'), data=diamonds) +\
    geom_density() +\
    scale_colour_manual(values=["coral", "chocolate", "cornsilk", "papayawhip", "blanchedalmond"])
ggplot(aes(x='wt', y='mpg', colour='factor(cyl)'), data=mtcars) +\
    geom_point() +\
    scale_colour_manual(values=["white", "grey", "black"])
